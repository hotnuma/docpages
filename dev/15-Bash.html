Helpers

  • Usefull functions

    appname=${0##*/}

error_exit()
{
    msg="$1"
    test "$msg" != "" || msg="an error occurred"
    printf "*** $msg\nabort...\n"
    exit 1
}

usage_exit()
{
    echo "*** usage :"
    echo "$appname -opt1"
    echo "$appname -opt2 bla"
    echo "abort..."
    exit 1
}

ask_confirm()
{
    while true; do
        read -p "$1" ret
        case $ret in
            [Yy]*)
                return 0; break
            ;;
            [Nn]*)
                return 1; break
            ;;
            *)
                return 0; break
            ;;
        esac
    done
}

    case "$1" in
        -dev)
        opt_dev=1
        ;;
        -o)
            test $# -gt 1 || error_exit "*** missing option"
            shift
            opt_name="$1"
        ;;
        *)
        opt_dev=0
        ;;
    esac
    shift
done
